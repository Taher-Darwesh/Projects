# balancing the dataset

# filter out fraud lines
fraud = df[df.isFraud ==1]
# filter out by type category
fraud_cash = fraud[fraud.type == 'CASH_OUT']
fraud_tran = fraud[fraud.type == 'TRANSFER']
# create random sample to match the min number of fraud lines in category
fraud_cash_sam = fraud_cash.sample(n=4097)
# concat two datasets to create a balanced dataset
fraud_total = pd.concat([fraud_tran,fraud_cash_sam])
# shuffle dataframe
fraud_df = fraud_total.sample(frac=1)

# filter out fraud lines
non_fraud = df[df.isFraud ==0]
# filter out by type category
non_fraud_cash_out = non_fraud[non_fraud.type == 'CASH_OUT']
non_fraud_cash_in = non_fraud[non_fraud.type == 'CASH_IN']
non_fraud_debit = non_fraud[non_fraud.type == 'DEBIT']
non_fraud_paym = non_fraud[non_fraud.type == 'PAYMENT']
non_fraud_tran = non_fraud[non_fraud.type == 'TRANSFER']
# create random sample to match fraud lines 
non_fraud_cash_out_sam = non_fraud_cash_out.sample(n=4097)
non_fraud_cash_in_sam = non_fraud_cash_in.sample(n=4097)
non_fraud_cash_debit_sam = non_fraud_debit.sample(n=4097)
non_fraud_cash_paym_sam = non_fraud_paym.sample(n=4097)
non_fraud_cash_tran_sam = non_fraud_tran.sample(n=4097)
# concat the datasets to create a balanced dataset
non_fraud_total = pd.concat([non_fraud_cash_out_sam,non_fraud_cash_in_sam,non_fraud_cash_debit_sam,non_fraud_cash_paym_sam,non_fraud_cash_tran_sam])
# shuffle dataframe
non_fraud_df = non_fraud_total.sample(frac=1)

# final dataframe
# concat the datasets to create a balanced dataset
df2 = pd.concat([non_fraud_df,fraud_df])
# shuffle dataframe
df2 = df2.sample(frac=1)
