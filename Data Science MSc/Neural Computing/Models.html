# creating the MLP NN model
class Net(nn.Module):
    def __init__(self):
        super(Net, self).__init__()
        # NN architecture 11-8-4-2
        self.layer_1 = nn.Linear(11, 8) 
        self.layer_2 = nn.Linear(8, 4)
        self.layer_out = nn.Linear(4, 2) 
        # activation function
        self.relu = nn.ReLU()
        # regularization on forward call
        self.dropout = nn.Dropout(p=0.01)
        # batch normalization
        self.batchnorm1 = nn.BatchNorm1d(8)
        self.batchnorm2 = nn.BatchNorm1d(4)
        
    def forward(self, inputs):
        # forward order
        x = self.relu(self.layer_1(inputs))
        x = self.batchnorm1(x)
        x = self.relu(self.layer_2(x))
        x = self.batchnorm2(x)
        x = self.dropout(x)
        x = self.layer_out(x)
        
        return x
# define binary accuracy function
def binary_acc(y_pred, y_test):
    # return sigmoid elements of input
    y_pred_tag = torch.round(torch.sigmoid(y_pred))
    correct_results_sum = (y_pred_tag == y_test).sum().float()
    acc = correct_results_sum/y_test.shape[0]
    acc = torch.round(acc * 100)
    
    return acc
